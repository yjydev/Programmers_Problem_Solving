## 네트워크    
- 풀이    

```javascript    
function solution(n, computers) {
    var answer = 0;
    var visited = [false];
    function dfs(i) {
        visited[i] = true;
        for (let j=0; j<n; j++) {
            if (computers[i][j] && !visited[j]) {
                dfs(j);
            }
        }
    }
    for (let i = 0; i <n; i++) {
        if (!visited[i]) {
            dfs(i);
            answer++;
        }
    }
    return answer;
}
```
