```python   
def solution(k, score):
    answer = []
    board = []
    for s in score:
        if len(board) < k :
            board.append(s)
            board.sort(reverse=True)
        else :
            for i in range(len(board)):
                if board[i] < s:
                    board.insert(i,s)
                    break
                elif board[i] == s:
                    board.insert(i+1,s)
                    break
            if len(board) > k:
                board.pop()
        answer.append(min(board))
    return answer
```
